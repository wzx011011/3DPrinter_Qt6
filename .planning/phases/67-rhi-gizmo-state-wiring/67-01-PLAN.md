       ```cpp
       QVector3D RhiViewportRenderer::computeGizmoCenter() const
       {
         const int selected = m_prepareScene.selectedSourceObjectIndex();
         if (selected < 0)
           return {}; // no selection - gizmo sits at origin
         const auto &batches = m_prepareScene.modelBatches();
         for (const auto &b : batches)
         {
           if (b.sourceObjectIndex == selected)
           {
             return QVector3D(
                 (b.bounds.minX + b.bounds.maxX) * 0.5f,
                 (b.bounds.minY + b.bounds.maxY) * 0.5f,
                 (b.bounds.minZ + b.bounds.maxZ) * 0.5f);
           }
         }
         return {}; // selected index not found in current batches
       }
       ```
